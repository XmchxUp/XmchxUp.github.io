# @Title: 存在重复元素 (Contains Duplicate)
# @Author: 1394466835@qq.com
# @Date: 2019-06-14 19:27:17
# @Runtime: 148 ms
# @Memory: 16.8 MB
class Solution(object):
    def containsDuplicate(self, nums):
        """
        :type nums: List[int]
        :rtype: bool
        """
        return len(nums) != len(set(nums))
