// @Title: 删除回文子序列 (Remove Palindromic Subsequences)
// @Author: 1394466835@qq.com
// @Date: 2021-06-27 19:10:10
// @Runtime: 0 ms
// @Memory: 1.9 MB
func removePalindromeSub(s string) int {
    if len(s) == 0 {
		return 0
	}

	for i := 0; i < len(s)/2; i++ {
		if s[i] != s[len(s)-i-1] {
			return 2
		}
	}
	
	return 1
}
