# @Title: 有效的括号 (Valid Parentheses)
# @Author: 1394466835@qq.com
# @Date: 2020-02-12 20:33:46
# @Runtime: 44 ms
# @Memory: 12.8 MB
class Solution:
    def isValid(self, s: str) -> bool:
        dic = {'{':'}', '[':']', '(':')', '1':'1'}
        stack = ['1']
        for c in s:
            if c in dic: stack.append(c)
            elif dic[stack.pop()] != c: return False
        return len(stack) == 1
