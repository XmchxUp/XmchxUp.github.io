// @Title: 有效的括号 (Valid Parentheses)
// @Author: 1394466835@qq.com
// @Date: 2019-05-03 20:33:24
// @Runtime: 8 ms
// @Memory: 8.5 MB
class Solution {
public:
    bool isValid(string s) {
        stack<char> stack;
        for (int i = 0; i < s.size(); i++) { 
            if (s[i] == '(' || s[i] == '[' || s[i] == '{')
                stack.push(s[i]);
            else {
                if (stack.size() == 0)
                    return false;
                
                char c = stack.top();
                stack.pop();
                
                char match;
                if (s[i] == ')')
                    match = '(';
                else if (s[i] == ']')
                    match = '[';
                else {
                    assert(s[i] == '}');
                    match = '{';
                }

                if (c != match)
                    return false;
            }
        }
        
        if (stack.size() != 0)
            return false;
        return true;
    }
};
