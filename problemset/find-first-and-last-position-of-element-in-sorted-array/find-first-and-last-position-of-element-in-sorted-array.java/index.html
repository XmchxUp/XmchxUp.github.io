// @Title: 在排序数组中查找元素的第一个和最后一个位置 (Find First and Last Position of Element in Sorted Array)
// @Author: 1394466835@qq.com
// @Date: 2021-05-19 14:09:20
// @Runtime: 0 ms
// @Memory: 41.7 MB
class Solution {
    public int[] searchRange(int[] nums, int target) {
		int[] result = new int[]{-1, -1};
        if (nums.length == 0) return result;
		int begin = 0, end = nums.length - 1;
		int mid;
		// 二分搜索找第一个位置
		while (begin < end) {
			mid = ((end - begin) >> 1) + begin;
			if (target > nums[mid]) {
				begin = mid + 1;
			} else {
				end = mid;
			}
		}
		if (target != nums[begin]) return result;
		result[0] = begin;

		// 二分搜索找第二个位置
		end = nums.length - 1;
		while (begin < end) {
			// 这里要处理 [8,10] target = 8
			mid = begin + ((end - begin) >> 1) + 1;
			if (target < nums[mid]) {
				end = mid - 1;
			} else {
				begin = mid;
			}
		}
		result[1] = begin;
		return result;
    }
}
