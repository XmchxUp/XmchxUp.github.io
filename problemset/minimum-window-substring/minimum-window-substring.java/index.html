// @Title: 最小覆盖子串 (Minimum Window Substring)
// @Author: 1394466835@qq.com
// @Date: 2021-05-18 06:40:04
// @Runtime: 5 ms
// @Memory: 38.1 MB
class Solution {
    public String minWindow(String s, String t) {
        int[] chars = new int[126];
        boolean[] flags = new boolean[126];

        t.chars().forEach(c -> {
            ++chars[c];
            flags[c] = true;
        });

        int minLen = Integer.MAX_VALUE;
        int minStart = 0;
        int start = 0, count = 0;
        int len = s.length();
        for (int end = 0; end < len; end++) {
            if (flags[s.charAt(end)]) {
                if (chars[s.charAt(end)]-- > 0) {
                    ++count;
                }

                while (count == t.length()) {
                    if (end - start + 1 < minLen) {
                        minLen = end - start + 1;
                        minStart = start;
                    }
                    if (flags[s.charAt(start)] && ++chars[s.charAt(start)] > 0) {
                        count--;
                    }
                    ++start;
                }
            }
        }
        return minLen == Integer.MAX_VALUE ? "" : s.substring(minStart, minStart+minLen);
    }
}
