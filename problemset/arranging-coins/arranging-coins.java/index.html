// @Title: 排列硬币 (Arranging Coins)
// @Author: 1394466835@qq.com
// @Date: 2021-07-15 21:02:20
// @Runtime: 8 ms
// @Memory: 35.6 MB
class Solution {
    public int arrangeCoins(int n) {
        int c = 0;
		while (n > c) {
            c = c + 1;
			n = n - c;
		}
		return c;
    }
}
