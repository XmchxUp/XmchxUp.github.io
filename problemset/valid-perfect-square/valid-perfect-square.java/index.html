// @Title: 有效的完全平方数 (Valid Perfect Square)
// @Author: 1394466835@qq.com
// @Date: 2021-05-28 07:31:33
// @Runtime: 0 ms
// @Memory: 35.2 MB
class Solution {
    public boolean isPerfectSquare(int num) {
		int left = 1, right = num;
		while (left <= right) {
			int mid = left + (right - left) / 2;
			int res = num / mid, tail = num % mid;
			if (tail == 0 && res == mid) {
				return true;
			} else if (res < mid) {
				right = mid - 1;
			} else {
				left = mid + 1;
			}
		}
		return false;
    }
}
