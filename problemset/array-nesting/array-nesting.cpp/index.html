// @Title: 数组嵌套 (Array Nesting)
// @Author: 1394466835@qq.com
// @Date: 2019-08-25 09:43:01
// @Runtime: 20 ms
// @Memory: 9.6 MB
class Solution {
public:
    int arrayNesting(vector<int>& nums) {
        size_t maxsize = 0;
        for (int i = 0; i < nums.size(); i++) {
            size_t size = 0;
            for (int k = i; nums[k] != -1; size++) {
                int tmp = nums[k];
                nums[k] = -1; 
                k = tmp;
            }
            maxsize = max(maxsize, size);
        }

        return maxsize;
    }
};
