// @Title: 数组嵌套 (Array Nesting)
// @Author: 1394466835@qq.com
// @Date: 2019-08-25 09:37:41
// @Runtime: 5 ms
// @Memory: 49.6 MB
class Solution {
    public int arrayNesting(int[] nums) {
        int max_count = 0;
        for (int i = 0; i < nums.length; i++) {
            int index = i, count = 0;
            while (index != -1 && nums[index] != -1) {
                int temp = index;
                index = nums[index];
                nums[temp] = -1;
                count++;
            }
            max_count = Math.max(max_count,count);
        }
        return max_count;
    }
}
