// @Title: 搜索旋转排序数组 II (Search in Rotated Sorted Array II)
// @Author: 1394466835@qq.com
// @Date: 2021-07-30 20:57:44
// @Runtime: 0 ms
// @Memory: 38 MB
class Solution {
    public boolean search(int[] nums, int target) {
		int l = 0, r = nums.length - 1;
		int mid;
		while (l <= r) {
			mid = (r - l) / 2 + l;
			if (target == nums[mid] || target == nums[l] || target == nums[r]) {
				return true;
			}

			if (nums[mid] > nums[l]) {
				// 左区间有序
				if (target >= nums[l] && target < nums[mid]) {
					r = mid - 1;
				} else {
					l = mid + 1;
				}
			} else if (nums[mid] < nums[r]) {
				// 右区间有序
				if (target > nums[mid] && target <= nums[r]) {
					l = mid + 1;
				} else {
					r = mid - 1;
				}
			} else {
				// 无法判断是否有序
				l++;
			}
		}
		return false;
    }
}
