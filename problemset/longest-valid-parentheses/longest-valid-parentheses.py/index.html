# @Title: 最长有效括号 (Longest Valid Parentheses)
# @Author: 1394466835@qq.com
# @Date: 2019-04-09 19:30:01
# @Runtime: 72 ms
# @Memory: 13.6 MB
class Solution:
    def longestValidParentheses(self, s: str) -> int:
        st, b = [], [0]*len(s)
        for i, val in enumerate(s):
            if val == '(':
                st.append(i)
            elif st:
                b[st.pop()], b[i] = 1, 1

        c, mc = 0, 0
        for i in b:
            if i:
                c += 1
            else:
                mc = max(c, mc)
                c = 0

        return max(c, mc)
