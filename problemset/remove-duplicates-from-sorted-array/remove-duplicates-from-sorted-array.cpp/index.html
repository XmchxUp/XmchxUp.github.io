// @Title: 删除有序数组中的重复项 (Remove Duplicates from Sorted Array)
// @Author: 1394466835@qq.com
// @Date: 2019-06-14 09:40:06
// @Runtime: 32 ms
// @Memory: 9.7 MB
class Solution {
public:
    int removeDuplicates(vector<int>& nums) {
        if (nums.size() == 0)
            return 0;
        int res = 1;
        int index = nextDifferentCharacterIndex(nums, 1);
        int i = 1;
        while (index != nums.size()) {
            res++;
            nums[i++] = nums[index];
            index = nextDifferentCharacterIndex(nums, index + 1);
        }

        return res;
    }
private:
    int nextDifferentCharacterIndex(const vector<int> & nums, int p) {
        for (; p < nums.size(); p++) {
            if (nums[p] != nums[p - 1])
                break;
        }
        return p;
    }
    
};
