// @Title: 组合总和 II (Combination Sum II)
// @Author: 1394466835@qq.com
// @Date: 2021-10-01 20:00:12
// @Runtime: 2 ms
// @Memory: 38.6 MB
class Solution {
    public List<List<Integer>> combinationSum2(int[] candidates, int target) {
        Arrays.sort(candidates);
        List<List<Integer>> ans = new ArrayList<>();
        List<Integer> cur = new ArrayList<>();
        dfs(candidates, ans, 0, target, cur);
        return ans;
    }

    private void dfs(int[] candidates, List<List<Integer>> ans, 
                     int s, int target, List<Integer> cur) {
        if (target == 0) {
            ans.add(new ArrayList<>(cur));
            return;
        }

        for (int i = s; i < candidates.length; i++) {
            if (candidates[i] > target) break;
            if (i > s && candidates[i - 1] == candidates[i]) continue;
            cur.add(candidates[i]);
            dfs(candidates, ans, i + 1, target - candidates[i], cur);
            cur.remove(cur.size() - 1);
        }
    }
}
