// @Title: 索引处的解码字符串 (Decoded String at Index)
// @Author: 1394466835@qq.com
// @Date: 2019-04-26 17:21:29
// @Runtime: 8 ms
// @Memory: 8.2 MB
class Solution {

public:
    /// Recursion
    /// Time Complexity: O(len(S)*logK)
    /// Space Complexity: O(logK)
    string decodeAtIndex(string S, int K) {
        int curl = 0;
        int repeat = 1;
        for (auto c : S) {
            if (isalpha(c)) {
                if (curl*repeat+1 == K) {
                    return string(1,c);
                }
                curl = curl * repeat + 1;
                repeat = 1;
            } else {
                repeat *= (c - '0');
                if ((long long)K <= (long long)curl * (long long)repeat) {
                    return decodeAtIndex(S, (K-1) % curl + 1);
                }
            }
        }
        
        return "";
    }
    
};
