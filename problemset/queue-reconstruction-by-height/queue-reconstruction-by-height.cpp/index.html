// @Title: 根据身高重建队列 (Queue Reconstruction by Height)
// @Author: 1394466835@qq.com
// @Date: 2019-04-01 17:39:23
// @Runtime: 60 ms
// @Memory: 9.9 MB
class Solution {
public:
    vector<pair<int, int>> reconstructQueue(vector<pair<int, int>>& people) {
        /*根据身高降序排序 如何相同在 按k升序排序*/
        auto rul = [](const pair<int,int>& p1, const pair<int,int>& p2)
                     { return p1.first == p2.first ? p1.second < p2.second : p1.first > p2.first;};
        sort(people.begin(), people.end(), rul);
        
        //身高矮的直接插入k的位置
        vector<pair<int, int>> res;
        for(auto& p: people)
        {
            res.insert(res.begin() + p.second, p);
        }
        return res;
    }
};
