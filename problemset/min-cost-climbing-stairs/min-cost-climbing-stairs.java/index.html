// @Title: 使用最小花费爬楼梯 (Min Cost Climbing Stairs)
// @Author: 1394466835@qq.com
// @Date: 2021-09-12 20:56:57
// @Runtime: 1 ms
// @Memory: 37.9 MB
class Solution {
    public int minCostClimbingStairs(int[] cost) {
        // 第i阶需要花费的最少体力
        int n = cost.length;
        int[] dp = new int[n + 1];    
        dp[0] = cost[0];
        dp[1] = cost[1];
        for (int i = 2; i < n; i++) {
            dp[i] = Math.min(dp[i - 1], dp[i - 2]) + cost[i];
        }
        return Math.min(dp[n - 1], dp[n - 2]);
    }
}
