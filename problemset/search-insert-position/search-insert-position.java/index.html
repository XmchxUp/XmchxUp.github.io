// @Title: 搜索插入位置 (Search Insert Position)
// @Author: 1394466835@qq.com
// @Date: 2021-10-03 08:05:04
// @Runtime: 0 ms
// @Memory: 38.1 MB
class Solution {
    public int searchInsert(int[] nums, int target) {
        int low = 0, high = nums.length;
        int mid;
        // [low, high)
        while (low < high) {
            mid = (high - low) / 2 + low;
            if (nums[mid] == target) return mid;
            if (nums[mid] > target) {
                high = mid;
            } else {
                low = mid + 1;
            }
        }

        return low;
    }
}
