// @Title: 至少是其他数字两倍的最大数 (Largest Number At Least Twice of Others)
// @Author: 1394466835@qq.com
// @Date: 2019-04-26 10:25:08
// @Runtime: 8 ms
// @Memory: 8.2 MB
class Solution {
    
/// Linear Scan
/// Time Complexity: O(n)
/// Space Complexity: O(1)
public:
    int dominantIndex(vector<int>& nums) {
        int maxNum = *max_element(nums.begin(), nums.end());
        int ret = -1;
        
        for (int i = 0; i < nums.size(); ++i) {
            if (nums[i] != maxNum) {
                if (2 * nums[i] > maxNum) 
                    return -1;
            } else {
                ret = i;
            }
        }
        
        return ret;
    }
};
