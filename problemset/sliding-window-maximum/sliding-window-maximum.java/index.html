// @Title: 滑动窗口最大值 (Sliding Window Maximum)
// @Author: 1394466835@qq.com
// @Date: 2020-02-29 16:23:55
// @Runtime: 15 ms
// @Memory: 47.8 MB
class Solution {
    public int[] maxSlidingWindow(int[] nums, int k) {
        if (nums == null || k == 0)
            return new int[0];
        int n = nums.length;
        int[] r = new int[n-k+1];
        int index = 0;
        Deque<Integer> q = new ArrayDeque<>();
        for (int i = 0; i < n; i++) {
            // remove numbers out of range k
            while (!q.isEmpty() && q.peek() < i - k + 1)
                q.poll();
            // remove smaller numbers in range k
            while (!q.isEmpty() && nums[q.peekLast()] < nums[i])
                q.pollLast();
            q.offer(i);
            if (i >= k-1) {
                r[index++] = nums[q.peek()];
            }
        }
        return r;
    }
}
