// @Title: 太平洋大西洋水流问题 (Pacific Atlantic Water Flow)
// @Author: 1394466835@qq.com
// @Date: 2021-09-19 21:58:59
// @Runtime: 5 ms
// @Memory: 39.8 MB
class Solution {
	// DFS O()
	public List<List<Integer>> pacificAtlantic(int[][] heights) {
		List<List<Integer>> ans = new ArrayList<>();

		int m = heights.length;
		int n = heights[0].length;

		boolean[][] p = new boolean[m][n];
		boolean[][] q = new boolean[m][n];

		for (int x = 0; x < n; x++) {
			dfsV2(p, x, 0, 0, heights);
			dfsV2(q, x, m - 1, heights[m - 1][x], heights);
		}

		for (int y = 0; y < m; y++) {
			dfsV2(p, 0, y, 0, heights);
			dfsV2(q, n - 1, y, 0, heights);
		}


		for (int i = 0; i < heights.length; i++) {
			for (int j = 0; j < heights[0].length; j++) {
				if (p[i][j] && q[i][j]) {
					ans.add(List.of(i, j));
				}
			}
		}
		return ans;
	}

	private void dfsV2(boolean[][] marked, int x, int y, int h, int[][] heights) {
		if (x < 0 || y < 0 || x == heights[0].length || y == heights.length) return;
		if (marked[y][x] || heights[y][x] < h) return;
		marked[y][x] = true;
		dfsV2(marked, x + 1, y, heights[y][x], heights);
		dfsV2(marked, x - 1, y, heights[y][x], heights);
		dfsV2(marked, x, y + 1, heights[y][x], heights);
		dfsV2(marked, x, y - 1, heights[y][x], heights);
	}
}
