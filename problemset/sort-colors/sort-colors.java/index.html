// @Title: 颜色分类 (Sort Colors)
// @Author: 1394466835@qq.com
// @Date: 2021-09-25 10:22:59
// @Runtime: 0 ms
// @Memory: 37 MB
class Solution {
    public void sortColors(int[] nums) {
        int rp = 0, wp = 0, bp = nums.length - 1;
        for (int i = 0; i <= bp; i++) {
            while (nums[i] == 2 && i < bp) {
                swap(nums, i, bp);
                bp--;
            }

            while (nums[i] == 0 && i > rp) {
                swap(nums, i , rp);
                rp++;
            }
        }
    }

    private void swap(int[] array, int i, int j) {
        int tmp = array[i];
        array[i] = array[j];
        array[j] = tmp;
    }
}
