// @Title: 猜数字大小 (Guess Number Higher or Lower)
// @Author: 1394466835@qq.com
// @Date: 2019-04-15 17:27:18
// @Runtime: 8 ms
// @Memory: 8 MB
// Forward declaration of guess API.
// @param num, your guess
// @return -1 if my number is lower, 1 if my number is higher, otherwise return 0
int guess(int num);

class Solution {
public:
    int guessNumber(int n) {
        //Binary Search
        int low = 1, high = n;
        while(low <= high)
        {
            int mid = low + (high - low) / 2;
            if(guess(mid) == 1)
            {
                low = mid + 1;
            }
            else if (guess(mid) == -1)
            {
                high = mid - 1;
            }
            else if (guess(mid) == 0)
            {
                return mid;
            }
        }
        return 0;
    }
};
