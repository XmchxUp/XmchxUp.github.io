// @Title: 01 矩阵 (01 Matrix)
// @Author: 1394466835@qq.com
// @Date: 2021-07-23 20:32:35
// @Runtime: 16 ms
// @Memory: 40.2 MB
class Solution {
    public int[][] updateMatrix(int[][] mat) {
        int m = mat.length;
		int n = mat[0].length;

		Queue<int[]> queue = new LinkedList<>();
		for (int i = 0; i < m; i++) {
			for (int j = 0; j < n; j++) {
				if (mat[i][j] == 0) {
					queue.offer(new int[]{i, j});
				} else {
					mat[i][j] = Integer.MAX_VALUE;
				}
			}
		}
		int[][] dires = {{-1, 0}, {1, 0}, {0, -1}, {0, 1}};
		while (!queue.isEmpty()) {
			int[] cell = queue.poll();
			for (int[] dire : dires) {
				int r = cell[0] + dire[0];
				int c = cell[1] + dire[1];
				if (r < 0 || r >= m || c < 0 || c >= n ||
						mat[r][c] <= mat[cell[0]][cell[1]] + 1) continue;
				queue.add(new int[]{r, c});
				mat[r][c] = mat[cell[0]][cell[1]] + 1;
			}
		}
		return mat;
    }
}
