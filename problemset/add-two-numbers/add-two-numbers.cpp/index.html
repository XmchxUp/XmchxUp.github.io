// @Title: 两数相加 (Add Two Numbers)
// @Author: 1394466835@qq.com
// @Date: 2019-01-25 11:07:20
// @Runtime: 48 ms
// @Memory: 1.7 MB
/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
    ListNode* addTwoNumbers(ListNode* l1, ListNode* l2) {
        int x = 0, y = 0;  //存储l1 l2当前的值
        int carry = 0; //存储进位值
        int sum = 0; //存储当前两个值的和
        ListNode* h = NULL, **t = &h;   //h用于返回的结果链表  t用于操作h链表
        
        while(l1 != NULL || l2 != NULL)
        {
            x = getValueAndMoveNext(l1);
            y = getValueAndMoveNext(l2);
            
            //存储当前和
            sum = carry + x + y;
            
            //创建新节点并挂上h链表 尾插法  初始化值为sum%10 一位数字
            ListNode * node = new ListNode(sum%10);
            *t = node;
            t = &(node->next);
            
            //更新进位值
            carry = sum / 10;
        }
        //所有执行完成 确保是否有进位
        if(carry > 0)
        {
            //创建新节点并挂上h链表  初始化值为sum%10 一位数字
            ListNode * node = new ListNode(carry%10);
            *t = node;
        }
        
        return h;
    }
private:
    //获取当前节点的值并移动到下一个节点
    int getValueAndMoveNext(ListNode* &l)
    {
        int x = 0;
        if (l != NULL)
        {
            x = l->val;
            l = l->next;
        }
        return x;
    }
};
